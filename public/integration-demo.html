üìã Integration Instructions

Method 1: Simple Script Tag (Recommended)

<!-- Add this before closing </body> tag -->
<script>
  // Optional: Configure the chatbot before loading
  window.nexusChatbotConfig = {
    position: 'bottom-right', // bottom-right, bottom-left, top-right, top-left
    theme: 'default',
    autoOpen: false,
    welcomeMessage: "Hello! How can I help you today?"
  };
</script>
<script src="https://your-domain.com/nexus-chatbot-widget.js"></script>

Method 2: Manual Initialization

<script src="https://your-domain.com/nexus-chatbot-widget.js"></script>
<script>
  // Initialize manually with custom config
  const chatbot = NexusChatbot.start({
    position: 'bottom-left',
    autoOpen: true,
    apiBaseUrl: 'https://your-custom-api.com'
  });
</script>

Method 3: WordPress Integration

// Add to your theme's functions.php
function add_nexus_chatbot() {
  ?>
  <script>
    window.nexusChatbotConfig = {
      position: 'bottom-right',
      theme: 'default'
    };
  </script>
  <script src="/js/nexus-chatbot-widget.js"></script>
  <?php
}
add_action('wp_footer', 'add_nexus_chatbot');

üîß API Methods

// Access the chatbot instance
const chatbot = NexusChatbot.widget;

// Open/close programmatically
chatbot.open();
chatbot.close();

// Send a message programmatically
chatbot.sendProgrammaticMessage("Hello from the website!");

// Update configuration
chatbot.updateConfig({ position: 'bottom-left' });

// Remove the widget
chatbot.destroy();
